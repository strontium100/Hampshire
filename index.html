let gameMode="fleet";

const shipTypeBonus={
  "装甲艦":{attack:1,defense:1.15,speed:0.9},
  "蒸気軍艦":{attack:1,defense:1,speed:1},
  "帆走軍艦":{attack:1,defense:0.9,speed:1.1},
  "輸送船":{attack:0,defense:1,speed:1}
};

const player={fleet:[]};
const cpu={fleet:[]};

function setMode(mode){
  gameMode=mode;
  alert("モード："+mode);
}

function createFleet(side){

  const baseShips=[
    {name:"開陽丸",type:"装甲艦",fire:90,defense:75,speed:45,hp:220},
    {name:"回天丸",type:"蒸気軍艦",fire:75,defense:70,speed:50,hp:180},
    {name:"蟠竜丸",type:"帆走軍艦",fire:60,defense:50,speed:80,hp:150}
  ];

  side.fleet=baseShips.map(s=>({...s,currentHP:s.hp}));

  if(gameMode==="escort"){
    const transport={name:"輸送船",type:"輸送船",fire:0,defense:30,speed:40,hp:160};
    side.fleet.push({...transport,currentHP:transport.hp});
  }
}

function startBattle(){

  createFleet(player);
  createFleet(cpu);

  let log="===戦闘開始===\n";

  for(let turn=1;turn<=5;turn++){

    log+="\n---ターン"+turn+"---\n";

    let all=[...player.fleet.map(s=>({...s,side:"player"})),
             ...cpu.fleet.map(s=>({...s,side:"cpu"}))];

    all.sort((a,b)=>
      (b.speed*shipTypeBonus[b.type].speed)-
      (a.speed*shipTypeBonus[a.type].speed)
    );

    for(let ship of all){

      if(ship.currentHP<=0) continue;

      let enemyFleet=ship.side==="player"?cpu.fleet:player.fleet;
      let alive=enemyFleet.filter(s=>s.currentHP>0);
      if(alive.length===0) break;

      let target;

      if(gameMode==="escort" && ship.side==="cpu"){
        let transport=alive.find(s=>s.type==="輸送船");
        if(transport && Math.random()<0.7){
          target=transport;
        }else{
          target=alive[Math.floor(Math.random()*alive.length)];
        }
      }else{
        target=alive[Math.floor(Math.random()*alive.length)];
      }

      let fire=ship.fire*shipTypeBonus[ship.type].attack;
      let defense=target.defense*shipTypeBonus[target.type].defense;
      let damage=Math.max(5,Math.floor(fire-defense*0.5));

      target.currentHP-=damage;

      log+=ship.name+" → "+target.name+" "+damage+"ダメージ\n";

      if(target.currentHP<=0){

        log+="💥"+target.name+"撃沈！\n";

        if(gameMode==="escort" && target.type==="輸送船"){
          log+="🚢輸送艦喪失！即敗北！\n";
          renderFleets();
          document.getElementById("battleLog").innerText=log;
          return;
        }

        if(target===enemyFleet[0]){
          log+="🚩旗艦喪失！即敗北！\n";
          renderFleets();
          document.getElementById("battleLog").innerText=log;
          return;
        }
      }
    }

    renderFleets();
  }

  log+="\n===戦闘終了===";
  document.getElementById("battleLog").innerText=log;
}

function renderFleets(){
  renderSide(player.fleet,"playerFleet");
  renderSide(cpu.fleet,"cpuFleet");
}

function renderSide(fleet,id){

  const container=document.getElementById(id);
  container.innerHTML="";

  fleet.forEach((ship,index)=>{

    let card=document.createElement("div");
    card.className="shipCard";

    if(index===0) card.classList.add("flagship");
    if(ship.type==="輸送船") card.classList.add("transport");
    if(ship.currentHP<=0){
      card.classList.add("sunk");
      for(let i=0;i<3;i++){
        let smoke=document.createElement("div");
        smoke.className="smoke";
        card.appendChild(smoke);
      }
    }

    let hpPercent=Math.max(0,(ship.currentHP/ship.hp)*100);
    let hpColor="green";
    if(hpPercent<50) hpColor="orange";
    if(hpPercent<25) hpColor="red";

    card.innerHTML+=
      (index===0?"🚩 ":"")+
      ""+ship.name+"
"+
      getShipSVG(ship.type)+
      "
HP:"+Math.max(0,ship.currentHP)+"/"+ship.hp+
      `
`;

    container.appendChild(card);
  });
}

    case "蒸気軍艦":
      return ``;

    case "帆走軍艦":
      return ``;

    case "輸送船":
      return ``;
  }
}
